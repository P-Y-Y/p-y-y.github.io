【题解】P2495 [SDOI2011]消耗战

作者: Sheep's Blog 日期: 2020-08-21 阅读时间:6.9 分钟 字数:1431 分类: 树论 虚树 题解 阅读量: -

Description：

给定一棵树，每条边有一定的代价，有mmm次询问，每次给出kik_iki​个关键点，求断开一些边，使111号节点与这些关键点不连通的最小代价。

数据范围：n<=250000,m<=500000,∑ki<=500000.n<=250000,m<=500000,\sum k_i<=500000.n<=250000,m<=500000,∑ki​<=500000.

Solution：

首先我们可以想到一个朴素的DP：

设dpidp_idpi​为使iii节点与其子树内的关键点不连通的最小代价。

设当前节点为uuu，枚举的子节点为vvv，连接u,vu,vu,v边的边权为w(u,v)w(u,v)w(u,v)，那么我们可以得到以下状态转移方程：

若vvv为关键点，dpu=dpu+min{dpv,w(u,v)}dp_u=dp_u+min\{dp_v,w(u,v)\}dpu​=dpu​+min{dpv​,w(u,v)}.

若vvv不为关键点，dpu=dpu+w(u,v)dp_u=dp_u+w(u,v)dpu​=dpu​+w(u,v).

这样我们就可以得到一个复杂度为O(mn)O(mn)O(mn)的dpdpdp做法了，然鹅这显然是不行的，我们得进一步优化。

观察数据范围，我们可以看出，其实每次询问的kkk都是比较稀疏分布的，故其实我们都做了许多无用功——我们多考虑了许多无用的节点。

那么我们考虑用一种更优的方法，忽略掉这些对答案没有什么实际影响的节点——用虚树来解决问题。

我们考虑对我们的答案有影响的，必定有关键点，但我们要在虚树上维护这些点之间的位置关系，所以，我们的虚树里面还要有他们之间的lcalcalca。

实际上，我们只需将这些关键点按dfsdfsdfs序排序，然后求出相邻两点的lcalcalca，把他们加入虚树中即可。

注意到多个点对的lcalcalca可能是相同的，我们这里考虑用一个单调栈来避免一个节点多次入栈的情况：

考虑我们用一个单调栈来维护虚树上一条链：

如果当前节点与栈顶元素的lcalcalca即为当前栈顶元素，说明当前节点与栈顶元素在一条链上，直接加入当前节点即可。

如果当前节点与栈顶元素的lcalcalca不为当前栈顶元素，说明当前节点与栈顶元素不在一条链上：

若lcalcalca比次顶元素的dfsdfsdfs序小，那我们就一直弹出栈顶元素，并向其后一位连边，直至lcalcalca的dfsdfsdfs序不小于栈顶元素的dfsdfsdfs序。

如果lcalcalca的dfsdfsdfs序等于次顶元素的dfsdfsdfs序，就说明lcalcalca已入栈（即为次顶元素），将栈顶元素与lcalcalca连边，弹出栈顶元素；否则，说明lcalcalca还未入栈，将栈顶元素与lcalcalca连边，弹出栈顶元素，再加入lcalcalca即可。

这样虚树大体上就建完了。

注意，多加入一个点并不会影响正确性，那么我们就顺便一开始把111号节点加入虚树中，最后在虚树上进行一开始我们提到的朴素dpdpdp即可。

时间复杂度O(∑klogn)O(\sum klogn)O(∑klogn)。

Code：

不开long long 见祖宗QAQ

#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int maxn=25e4+111,inf=1e9+7;
int N,M,K,tot,tim,cnt;
int H[maxn],h[maxn],A[maxn],dfn[maxn],fa[maxn][21],fw[maxn][21],dep[maxn],sta[maxn];
ll dp[maxn]; 
bool book[maxn];
struct edge {
	int n,v,w;
}E[maxn<<1],e[maxn<<1];
void addE(int u,int v,int w)
{
	E[++tot]=(edge){H[u],v,w};
	H[u]=tot;
}
void adde(int u,int v,int w)
{
	e[++tot]=(edge){h[u],v,w};
	h[u]=tot;
}
void dfs(int po,int step)
{
	dfn[po]=++tim;dep[po]=step;
	for(int tt=1;(1<<tt)<=step;tt++) 
		fa[po][tt]=fa[fa[po][tt-1]][tt-1],fw[po][tt]=min(fw[po][tt-1],fw[fa[po][tt-1]][tt-1]);
	for(int i=H[po];i;i=E[i].n)
	{
		int v=E[i].v;
		if(!dfn[v]) fa[v][0]=po,fw[v][0]=E[i].w,dfs(v,step+1);
	}
}
int lca(int u,int v)
{
	if(dep[u]<dep[v]) swap(u,v);
	for(int i=18;i>=0;i--) if(dep[fa[u][i]]>=dep[v]) u=fa[u][i];
	if(u==v) return u;
	for(int i=18;i>=0;i--) if(fa[u][i]!=fa[v][i]) u=fa[u][i],v=fa[v][i];
	return fa[u][0]; 
}
int getw(int u,int v)
{
	int sum=inf;
	for(int i=18;i>=0;i--) if(dep[fa[u][i]]>=dep[v]) sum=min(sum,fw[u][i]),u=fa[u][i];
	return sum; 
}
bool cmp(int a,int b)
{
	return dfn[a]<dfn[b];
} 
void T_dp(int po)
{
	dp[po]=0;
	for(int i=h[po];i;i=e[i].n)
	{
		int v=e[i].v,w=e[i].w;
		T_dp(v);
		if(book[v]) dp[po]=dp[po]+w;
		else dp[po]=dp[po]+min(1ll*w,dp[v]); 
	}
}
#define gc (getchar())
#define gs (ch<'0'||ch>'9')
void ri(int &x)
{
	x=0;char ch=gc;
	while( gs ) ch=gc;
	while(!gs ) x=x*10+ch-'0',ch=gc;
}
void wt(ll x)
{
	if(x>9) wt(x/10);
	putchar(x%10+'0');
}
int main()
{
	ri(N);
	int u,v,w;
	for(int i=1;i< N;i++) ri(u),ri(v),ri(w),addE(u,v,w),addE(v,u,w);
	dfs(1,1);
	ri(M);
	while(M--)
	{
		ri(K);
		for(int i=1;i<=K;i++) ri(A[i]),book[A[i]]=1;
		sort(A+1,A+K+1,cmp);
		cnt=0,tot=0;
		sta[++cnt]=1,h[1]=0;
		for(int i=1;i<=K;i++)
			if(A[i]!=1)
			{
				int tt=lca(A[i],sta[cnt]);
				if(tt==sta[cnt]) sta[++cnt]=A[i],h[A[i]]=0;
				else {	
					while(dfn[sta[cnt-1]]>dfn[tt]) w=getw(sta[cnt],sta[cnt-1]),adde(sta[cnt-1],sta[cnt],w),sta[cnt--]=0;
					if(sta[cnt-1]!=tt) 
						w=getw(sta[cnt],tt),h[tt]=0,adde(tt,sta[cnt],w),sta[cnt]=tt;
					else 
						w=getw(sta[cnt],tt),adde(tt,sta[cnt],w),sta[cnt--]=0;
					sta[++cnt]=A[i],h[A[i]]=0;
				} 
			}	
		while(cnt>1) w=getw(sta[cnt],sta[cnt-1]),adde(sta[cnt-1],sta[cnt],w),sta[cnt--]=0;
		T_dp(1);
		wt(dp[1]),putchar('\n');
		for(int i=1;i<=K;i++) book[A[i]]=0;
	}
	return 0;
}
