【题解】P4887 【模板】莫队二次离线（第十四分块(前体)）

作者: Sheep's Blog 日期: 2020-08-24 阅读时间:5.6 分钟 字数:1028 分类: 莫队 题解 阅读量: -

Description

给定一个序列{Ai}\{A_i\}{Ai​}和kkk，有mmm次询问，每次询问区间[l,r][l,r][l,r]内满足l<=i<j<=rl<=i<j<=rl<=i<j<=r且popcount(Ai⊕Aj)==kpopcount(A_i \oplus A_j)==kpopcount(Ai​⊕Aj​)==k的二元组(i,j)(i,j)(i,j)的个数。

n,m<=1e5,0<=Ai,k<16384n,m<=1e5,0<=A_i,k<16384n,m<=1e5,0<=Ai​,k<16384

Solution

显然，如果我们直接用普通莫队的方法进行转移的话，复杂度是较劣的，为O(MNC(14,k))O(M\sqrt NC(14,k))O(MN​C(14,k)).

那么这一题显然就过不去惹QAQ

我们注意到，这样做复杂度较劣的原因是我们每次莫队转移的复杂度太高，那么有没有什么方法能够降低转移的复杂度呢？

这里 lxl 大佬提出了一种做法——二次离线，即将我们要转移的地方离线下来，然后再返回去统计答案。

比如说，在这道题中：

将区间[l,r][l,r][l,r]转移成[l,r+k][l,r+k][l,r+k]的过程中，对于x∈[r+1,r+k]x∈[r+1,r+k]x∈[r+1,r+k]，设f(x,l,r)f(x,l,r)f(x,l,r)为位置xxx对区间[l,r][l,r][l,r]的贡献，那么我们可以得到：

f(x,l,x−1)=f(x,1,x−1)−f(x,1,l−1)f(x,l,x-1)=f(x,1,x-1)-f(x,1,l-1)f(x,l,x−1)=f(x,1,x−1)−f(x,1,l−1)，那么对于f(x,1,x−1)f(x,1,x-1)f(x,1,x−1)我们直接预处理即可，f(x,1,l−1)f(x,1,l-1)f(x,1,l−1)需要再次离线下来求，这样我们的转移的复杂度就可以大大降低了。

时间复杂度为O(MN+NC(14,k)O(M\sqrt N+NC(14,k)O(MN​+NC(14,k))，但实际跑不满，可以通过此题。

Code

实现细节有点多，注意对于每种L、R的转移分类讨论。

#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;
const int maxn=1e5+111,maxm=1e5+111,maxnum=17777;
int N,M,K;
int A[maxn],B[maxn],C[maxnum],D[maxnum];
long long pre1[maxn],pre2[maxn],asr[maxm];
vector<int>buc;
struct query {
	int l,r,id;
	long long ans;
	bool operator < (const query b) const{return B[l]!=B[b.l]?B[l]<B[b.l]:B[l]&1?r<b.r:r>b.r;} 
}Q[maxm];
struct node {
	int l,r,id,opt;
};
vector<node>E[maxn]; 
#define gc (getchar())
#define gs (ch<'0'||ch>'9')
void ri(int &x)
{
	x=0;char ch=gc;
	while( gs ) ch=gc;
	while(!gs ) x=x*10+ch-'0',ch=gc;
}
void wt(long long x)
{
	if(x>9) wt(x/10);
	putchar(x%10+'0');
}
int main()
{
	ri(N),ri(M),ri(K);
	int block=sqrt(N);
	for(int i=1;i<=N;i++) ri(A[i]),B[i]=(i+block-1)/block;
	for(int i=1;i<=M;i++) ri(Q[i].l),ri(Q[i].r),Q[i].id=i;
	sort(Q+1,Q+M+1);
	for(int i=0;i< 16384;i++) if(__builtin_popcount(i)==K) buc.push_back(i);
	for(int i=1;i<=N;i++) 
	{
		pre1[i]=C[A[i]];
		for(int j=0;j< (int)buc.size();j++) C[A[i]^buc[j]]++;	
		pre2[i]=C[A[i]];
	} 
	int L=1,R=0;
	for(int i=1;i<=M;i++)
	{
		if(R< Q[i].r) E[L-1].push_back((node){R+1,Q[i].r,i,-1});
		while(R< Q[i].r) Q[i].ans+=pre1[++R];
		if(R> Q[i].r) E[L-1].push_back((node){Q[i].r+1,R,i, 1});
		while(R> Q[i].r) Q[i].ans-=pre1[R--];
		if(L< Q[i].l) E[R].push_back((node){L,Q[i].l-1,i,-1});
		while(L< Q[i].l) Q[i].ans+=pre2[L++];
		if(L> Q[i].l) E[R].push_back((node){Q[i].l,L-1,i, 1});
		while(L> Q[i].l) Q[i].ans-=pre2[--L]; 
	}
	memset(C,0,sizeof(C));
	for(int i=1;i<=N;i++)
	{
		for(int j=0;j< (int)buc.size();j++) C[A[i]^buc[j]]++;
		for(int j=0;j< (int)E[i].size();j++)
		{
			int id=E[i][j].id,opt=E[i][j].opt;
			for(int k=E[i][j].l;k<=E[i][j].r;k++)
				Q[id].ans+=opt*C[A[k]];
		}	
	}
	for(int i=1;i< M;i++) Q[i+1].ans+=Q[i].ans;
	for(int i=1;i<=M;i++) asr[Q[i].id]=Q[i].ans; 
	for(int i=1;i<=M;i++) wt(asr[i]),putchar('\n');
 	return 0;
}
