多项式模板篇

FFT

#include <iostream>
#include <cstdio>
#include <cmath>
#define gc (getchar())
#define gs (ch<'0'||ch>'9')
using namespace std;
const int maxn=2e6+111;
const double Pi=acos(-1.0);
int N,M;
struct cp
{
	double x,y;
	cp (double xx=0,double yy=0) {x=xx,y=yy;}
}A[maxn],B[maxn];
cp operator + (cp a,cp b) {return cp(a.x+b.x,a.y+b.y);}
cp operator - (cp a,cp b) {return cp(a.x-b.x,a.y-b.y);}
cp operator * (cp a,cp b) {return cp(a.x*b.x-a.y*b.y,a.x*b.y+a.y*b.x);}
void FFT(int tot,cp *a,int ty)
{
	if(tot==1) return;
	cp a1[(tot>>1)+7],a2[(tot>>1)+7];
	for(int i=0;i<=tot;i+=2)
		a1[i>>1]=a[i],a2[i>>1]=a[i+1];
	FFT(tot>>1,a1,ty);
	FFT(tot>>1,a2,ty);
	cp wn=cp(cos(2.0*Pi/tot),ty*sin(2.0*Pi/tot)),w=cp(1,0);
	for(int i=0;i<(tot>>1);i++,w=w*wn)
		a[i]=a1[i]+w*a2[i],
		a[i+(tot>>1)]=a1[i]-w*a2[i];
}
void ri(int &x)
{
	x=0;char ch=gc;
	while( gs ) ch=gc;
	while(!gs ) x=x*10+ch-'0',ch=gc;
}
void wt(int x)
{
	if(x<0) putchar('-'),x*=-1;
	if(x>9) wt(x/10);
	putchar(x%10+'0');
}
int main()
{
	ri(N),ri(M);
	int x;
	for(int i=0;i<=N;i++) 
		ri(x),A[i].x=x;
	for(int i=0;i<=M;i++) 
		ri(x),B[i].x=x;
	int tot=1;
	while(tot<N+M+2) tot<<=1; 
	FFT(tot,A,1);
	FFT(tot,B,1);
	for(int i=0;i<=tot;i++) A[i]=A[i]*B[i];
	FFT(tot,A,-1);
	for(int i=0;i<=N+M;i++) wt((int)(A[i].x/tot+0.5)),putchar(' ');
	return 0;
}

NTT

#include <iostream>
#include <cstdio>
using namespace std;
const int maxn=3e6+111,g=3,mo=998244353;
int N,M,tot=1;
int rev[maxn];
int ksm(int x,int tt)
{
	int res=1;
	while(tt)
	{
		if(tt&1) res=1ll*res*x%mo;
		x=1ll*x*x%mo;
		tt>>=1;
	}
	return res;
}
void NTT(int *a,int ty)
{
	for(int i=0;i< tot;i++) if(i<rev[i]) swap(a[i],a[rev[i]]);
	for(int len=2;len<=tot;len<<=1)
	{
		int tmp=ksm(g,(mo-1)/len);
		if(ty==-1) tmp=ksm(tmp,mo-2);
		for(int L=0;L< tot;L+=len)
		{
			int w=1;
			for(int k=L;k< L+(len>>1);k++,w=1ll*w*tmp%mo)
			{
				int x=a[k],y=1ll*w*a[k+(len>>1)]%mo;
				a[k]=(x+y)%mo,a[k+(len>>1)]=(x-y+mo)%mo;
			}	
		}	
	} 
}
int A[maxn],B[maxn];
#define gc (getchar())
#define gs (ch<'0'||ch>'9')
void r(int &x)
{
	x=0;char ch=gc;
	while( gs ) ch=gc;
	while(!gs ) x=x*10+ch-'0',ch=gc;
}
void wt(int x)
{
	if(x>9) wt(x/10);
	putchar(x%10+'0');
}
int main()
{
	r(N),r(M);
	for(int i=0;i<=N;i++) r(A[i]);
	for(int i=0;i<=M;i++) r(B[i]);
	while(tot<=N+M) tot<<=1;
	for(int i=0;i< tot;i++) rev[i]=(rev[i>>1]>>1)|((i&1)?tot>>1:0);
	NTT(A,1);NTT(B,1);
	for(int i=0;i< tot;i++) A[i]=1ll*A[i]*B[i]%mo;
	NTT(A,-1);
	int inv=ksm(tot,mo-2);
	for(int i=0;i<=N+M;i++) wt(1ll*A[i]*inv%mo),putchar(' ');
	return 0;
}

#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int maxn=(1<<18)+1,mo=998244353;
int N;
int A[maxn],B[maxn],a[maxn],b[maxn];
#define gc (getchar())
#define gs (ch<'0'||ch>'9')
void ri(int &x)
{
	x=0;int f=1;char ch=gc;
	while( gs ) {if(ch=='-') f=-1;ch=gc;}
	while(!gs ) x=x*10+ch-'0',ch=gc;
	x*=f;
}
void wt(int x)
{
	if(x>9) wt(x/10);
	putchar(x%10+'0');
}
void read()
{
	ri(N);
	N=1<<N;
	for(int i=0;i< N;i++) ri(A[i]),A[i]=(A[i]+mo)%mo;
	for(int i=0;i< N;i++) ri(B[i]),B[i]=(B[i]+mo)%mo;
}
void init()
{
	memcpy(a,A,N+7<<2);
	memcpy(b,B,N+7<<2);
}
void output()
{
	for(int i=0;i< N;i++) wt(a[i]),putchar(' ');
	putchar('\n');
}
void upd(int &x,int y)
{
	y=y<0?y+mo:y;
	x=(x+y)%mo;
}
void OR(int *f,int x)
{
	for(int k=2;k<=N;k<<=1)
		for(int i=0;i< N;i+=k)
			for(int j=i;j< i+k/2;j++)
				upd(f[j+k/2],f[j]*x);
}
void AND(int *f,int x)
{
	for(int k=2;k<=N;k<<=1)
		for(int i=0;i< N;i+=k)
			for(int j=i;j< i+k/2;j++)
				upd(f[j],f[j+k/2]*x);
}
void XOR(int *f,int x)
{
	for(int k=2;k<=N;k<<=1)
		for(int i=0;i< N;i+=k)
			for(int j=i;j< i+k/2;j++)
			{
				int aa=f[j],bb=f[j+k/2];
				f[j]=(aa+bb)%mo,f[j+k/2]=(aa-bb+mo)%mo;
				f[j]=1ll*f[j]*x%mo,f[j+k/2]=1ll*f[j+k/2]*x%mo;
			}	
}
void MUL()
{
	for(int i=0;i< N;i++) a[i]=(1ll*a[i]*b[i])%mo;
}
int ksm(int x,int tt)
{
	int res=1;
	while(tt)
	{
		if(tt&1) res=1ll*res*x%mo;
		x=1ll*x*x%mo;
		tt>>=1; 
	}
	return res;
}
int main()
{
	read();
	init(); OR(a,1);  OR(b,1);  MUL(); OR(a,-1);  output();
	init(); AND(a,1); AND(b,1); MUL(); AND(a,-1); output();
	init(); XOR(a,1); XOR(b,1); MUL(); XOR(a,ksm(2,mo-2)); output();
	return 0;
}
